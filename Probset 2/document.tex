	Given 
	\begin{equation*}
	u ''(t) = f(t) 
	\end{equation*}
	we apply the Fundamental Theorem of Calculus to get,
	\begin{equation*}
	\int_0^t u ''(t) = u'(t) = c_{1} + \int_0^t f(t)dt 
	\end{equation*}
	where $c_1$ is determined by the initial condition. We can rewrite this as a definite integral
	\begin{equation*}
	\begin{split}
	u'(t) &= c_{1} + \int_0^t f(x)dx \\
	&= c_{1} + F(t)
	\end{split}
	\end{equation*}
	where $F(t) = \int_0^t f(x)dx$. Integrating one more time using the Fundamental Theorem of Calculus,
	\begin{equation*}'
	\begin{split}
	\int_0^t u '(t) = u(t) &= \int_0^t \Big( c_{1} + F(t) \Big)dt \\
	u(t) &= c_{2} + c_{1}t + \int_0^t F(s) ds 
	\end{split}
	\end{equation*}
	where $F(s) = \int_0^s f(x)dx$ 
	
	\paragraph{10.1b}
	
	Show $u(t)=c_{1}+c_{2}t+\int_{0}^{t}(t-s)f(s)ds$
	
	From a), $u(t)=c_{1}+c_{2}t+\int_{0}^{t}F(s)ds$
	
	It suffice to show that $\int_{0}^{t}F(s)ds=\int_{0}^{t}(t-s)f(s)ds$
	
	Using integration by parts, $\int udv=uv-\int vdu$
	
	Let u=F(s) and du=dF(s), then dv = ds and v =s.
	
	$\int_{0}^{t}F(s)ds=sF(s)|_{0}^{t}-\int_{0}^{t}sdF(s)=tF(t)-0F(0)-\int_{0}^{t}sdF(s)=tF(t)-\int_{0}^{t}sf(s)ds=\int_{0}^{t}tf(s)ds-\int_{0}^{t}sf(s)ds=\int_{0}^{t}(t-s)f(s)ds.$

	\paragraph{11.6}
	Prove that the Jacobi iterative method for solving the linear system $Ax=b$ converges if the matrix $A$ is diagonally dominant by rows. \\ \\
	\textbf{Proof.} 
	By definition, A is diagonally dominant if $|a_{ii}| > \sum_{j \neq i} |a_{ij}|$. In other words, we can define r
	\begin{equation*}
	r = \sum_{j \neq i} \frac{|a_{ij}|}{|a_{ii}|} < 1
	\end{equation*}
	
	%Reference: http://www.math.tamu.edu/~pasciak/classes/639/l2.pdf
	A linear iteration method converges for any starting iterate and
	any right hand side provided that matrix norm $|| G || < 1$.
	Recall that Jacobi iteration scheme is given by
	\begin{equation*}
	x^{k+1} = -D^{-1}(L+U)x^{(k)} + D^{-1}b
	\end{equation*}
	where $A = D + L + U$. 
	
	Therefore, for convergence we need $|| G || = ||-D^{-1}(L+U)x^{(k)}|| < 1$. 
	The $\infty$-norm of a matrix $M$ is given by 
	\begin{equation*}
	||M||_{\infty} = \max_{1\leq i\leq n} \sum_{j=1}^{n} |m_{ij}|
	\end{equation*}
	
	If A is diagonally dominant by rows, then the following must be true
	\begin{equation*}
	||-D^{-1}(L+U)x^{(k)}||_{\infty} = \max \sum_{j \neq i} \frac{|a_{ij}|}{|a_{ii}|} \leq r < 1
	\end{equation*}
	Therefore, the Jacobi iterative method for solving the linear systems converges if matrix $A$ is diagonally dominant by rows.
	
	\paragraph{11.7}
	
	\def\defeq{\mathrel{\mathop:}=}
	
	Let \emph{B} be any $n\times n$ complex matrix with
	zero diagonal entries which we write in the form \emph{B}=\emph{L}+\emph{U},
	where \emph{L} and \emph{U} are, respectively, strictly lower and
	strictly upper triangular matrices. If 
	\[
	L_{\omega}\defeq(I-\omega L)^{-1}(\omega U+(1-\omega)I),
	\]
	
	
	then for any real or complex number $\omega$,
	
	\[
	\rho(L_{\omega})\ge|\omega-1|,
	\]
	
	
	with equality holding only if all eigenvalues of $L_{\omega}$ are
	of modulus $|\omega-1|$.
	
	\textbf{Proof.} Let $\phi(\lambda)\defeq\det(\lambda I-L_{\omega})$ be
	the characteristic polynomial of $L_{\omega}$. Since L is strictly
	lower triangular, then $(I-\omega L)$ is nonsingular, and as $\det(I-\omega L)=1$,
	then
	
	\[
	\phi(\lambda)=\det(I-\omega L)\cdot\det(\lambda I-L_{\omega})=\det\{(\lambda+\omega-1)I-\omega\lambda L-\omega U\}.
	\]
	
	
	The constant term term $\mu$ of $\phi(\lambda)$, the product of
	the negatives of the eigenvalues of $L_{\omega}$, is obtained by
	setting $\lambda=0$ in this expression. Thus,
	
	\[
	\mu=\prod_{i=1}^{n}(-\lambda_{i}(\omega))=\det\{(\omega-1)I-\omega U\}=(\omega-1)^{n}.
	\]
	
	
	Therefore,
	
	\[
	\rho(L_{\omega})=\max_{1\le i\le n}|\lambda_{i}(\omega)|\ge|\omega-1|,
	\]
	
	
	with equality possible only if all eigenvalues of $L_{\omega}$ are
	modulus $|\omega-1|$, which completes the proof. If we wish to determine
	the \emph{real} value of $\omega$ which minimizes $\rho(L_{\omega})$,
	then the above result of Kahan shows that we need only consider the
	open interval $0<\omega<2$ for values of $\omega$ such that $L_{\omega}$
	is convergent.
